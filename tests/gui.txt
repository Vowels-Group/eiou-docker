# GUI Integration Instructions for All Systems
# This links the local src/wallet.html with the container's /var/www/html/index.html

# Windows (PowerShell/Command Prompt)
docker run -d --name Bob --network eiou-network -v "${PWD}\src\wallet.html:/var/www/html/index.html" -e QUICKSTART=Bob eioud

# Linux/macOS (Bash/Zsh)
docker run -d --name Bob --network eiou-network -v "$(pwd)/src/wallet.html:/var/www/html/index.html" -e QUICKSTART=Bob eioud

# Alternative absolute path examples:
# Windows: -v "C:\path\to\your\eiou\src\wallet.html:/var/www/html/index.html"
# Linux: -v "/home/user/eiou/src/wallet.html:/var/www/html/index.html"
# macOS: -v "/Users/user/eiou/src/wallet.html:/var/www/html/index.html"
