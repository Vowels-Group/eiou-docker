# GUI Integration Instructions for All Systems
# This document provides instructions for building and running EIOU containers with GUI access

## Build Instructions

# Build the Docker image (run this first)
docker build -f eiou.dockerfile -t eiou/eiou .

# Create the network (only need to do this once per docker installation)
docker network create -d bridge eiou-network



## Container Setup Options


### Option 1: Bob with Volume Mount (for development/testing)
# This links the local src/ folder with the container's /var/www/app/eiou/src/ folder
# Use this when you want to modify the GUI locally and see changes immediately


# Windows (PowerShell/Command Prompt) 
docker run -d --restart unless-stopped --name Alice --network eiou-network -v "${PWD}\files\src\:/app/eiou/src/" -e QUICKSTART=Alice eiou/eiou
docker run -d --restart unless-stopped --name Bob --network eiou-network -v "${PWD}\files\src\:/app/eiou/src/" -e QUICKSTART=Bob eiou/eiou
docker run -d --restart unless-stopped --name Carol --network eiou-network -v "${PWD}\files\src\:/app/eiou/src/" -e QUICKSTART=Carol eiou/eiou
docker run -d --restart unless-stopped --name Daniel --network eiou-network -v "${PWD}\files\src\:/app/eiou/src/" -e QUICKSTART=Daniel eiou/eiou

# Linux/macOS (Bash/Zsh)
docker run -d --restart unless-stopped --name Alice --network eiou-network -v "$(pwd)/files/src/:/app/eiou/src/" -e QUICKSTART=Alice eiou/eiou
docker run -d --restart unless-stopped --name Bob --network eiou-network -v "$(pwd)/files/src/:/app/eiou/src/" -e QUICKSTART=Bob eiou/eiou
docker run -d --restart unless-stopped --name Carol --network eiou-network -v "$(pwd)/files/src/:/app/eiou/src/" -e QUICKSTART=Carol eiou/eiou
docker run -d --restart unless-stopped --name Daniel --network eiou-network -v "$(pwd)/files/src/:/app/eiou/src/" -e QUICKSTART=Daniel eiou/eiou

# Alternative absolute path examples (GUI is served from /app/eiou/www/gui/):
# Windows: -v "C:\path\to\your\eiou\files\root\www\gui\index.html:/app/eiou/www/gui/index.html"
# Linux: -v "/home/user/eiou/files/root/www/gui/index.html:/app/eiou/www/gui/index.html"
# macOS: -v "/Users/user/eiou/files/root/www/gui/index.html:/app/eiou/www/gui/index.html"


### Option 2: Alice, Carol, and Daniel (standard containers)
# These containers use the built-in index.html from the Docker image
# No volume mounting needed - they use the default GUI

# Windows (PowerShell/Command Prompt)
docker run -d --restart unless-stopped --name Alice --network eiou-network -e QUICKSTART=Alice eiou/eiou
docker run -d --restart unless-stopped --name Carol --network eiou-network -e QUICKSTART=Carol eiou/eiou
docker run -d --restart unless-stopped --name Daniel --network eiou-network -e QUICKSTART=Daniel eiou/eiou

# Linux/macOS (Bash/Zsh)
docker run -d --restart unless-stopped --name Alice --network eiou-network -e QUICKSTART=Alice eiou/eiou
docker run -d --restart unless-stopped --name Carol --network eiou-network -e QUICKSTART=Carol eiou/eiou
docker run -d --restart unless-stopped --name Daniel --network eiou-network -e QUICKSTART=Daniel eiou/eiou

## Accessing the GUI

# After containers are running, access the GUI via:
# - Bob (with volume mount): http://localhost:8080 (if port mapping is added)
# - Alice, Carol, Daniel: Use the HTTP addresses shown in container logs

# To see container information and HTTP addresses:
docker logs Alice
docker logs Bob
docker logs Carol
docker logs Daniel

# To access the GUI directly from within a container:
docker exec -it Bob bash
# Then navigate to /app/eiou/src/gui/ and view index.html

## Notes

# - Bob's container with volume mount is ideal for GUI development
# - Alice, Carol, and Daniel use the standard built-in GUI
# - All containers automatically generate wallets and setup when QUICKSTART is used
# - The network 'eiou-network' allows containers to communicate with each other
