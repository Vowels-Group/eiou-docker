# create the network (only need to do this once per docker)
docker network create -d bridge demo-net

# Remove any existing containers
powershell -Command "if (docker ps -aq) { docker rm -f $(docker ps -aq) } else { echo 'No containers to remove' }"

# Build base image
docker build -f eiou.dockerfile -t eiou .

# Start containers
docker run -d  --network=demo-net --name httpA -e QUICKSTART=httpA eiou/eiou
docker run -d  --network=demo-net --name httpB -e QUICKSTART=httpB eiou/eiou
docker run -d  --network=demo-net --name httpC -e QUICKSTART=httpC eiou/eiou
docker run -d  --network=demo-net --name httpD -e QUICKSTART=httpD eiou/eiou
docker run -d  --network=demo-net --name httpE -e QUICKSTART=httpE eiou/eiou
docker run -d  --network=demo-net --name httpF -e QUICKSTART=httpF eiou/eiou
docker run -d  --network=demo-net --name httpG -e QUICKSTART=httpG eiou/eiou
docker run -d  --network=demo-net --name httpH -e QUICKSTART=httpH eiou/eiou
docker run -d  --network=demo-net --name httpI -e QUICKSTART=httpI eiou/eiou
docker run -d  --network=demo-net --name httpJ -e QUICKSTART=httpJ eiou/eiou
sleep 5

# add contacts
docker exec httpA eiou add http://httpB httpB 0.1 1000 USD
docker exec httpB eiou add http://httpA httpA 0.1 1000 USD
docker exec httpB eiou add http://httpC httpC 0.1 1000 USD
docker exec httpC eiou add http://httpB httpB 0.1 1000 USD
docker exec httpC eiou add http://httpD httpD 0.1 1000 USD
docker exec httpD eiou add http://httpC httpC 0.1 1000 USD
docker exec httpD eiou add http://httpE httpE 0.1 1000 USD
docker exec httpE eiou add http://httpD httpD 0.1 1000 USD
docker exec httpE eiou add http://httpF httpF 0.1 1000 USD
docker exec httpF eiou add http://httpE httpE 0.1 1000 USD
docker exec httpF eiou add http://httpG httpG 0.1 1000 USD
docker exec httpG eiou add http://httpF httpF 0.1 1000 USD
docker exec httpG eiou add http://httpH httpH 0.1 1000 USD
docker exec httpH eiou add http://httpG httpG 0.1 1000 USD
docker exec httpH eiou add http://httpI httpI 0.1 1000 USD
docker exec httpI eiou add http://httpH httpH 0.1 1000 USD
docker exec httpI eiou add http://httpJ httpJ 0.1 1000 USD
docker exec httpJ eiou add http://httpI httpI 0.1 1000 USD

# direct transaction
docker exec httpA eiou send http://httpB 100 USD

docker exec httpA eiou send http://httpC 50 USD
