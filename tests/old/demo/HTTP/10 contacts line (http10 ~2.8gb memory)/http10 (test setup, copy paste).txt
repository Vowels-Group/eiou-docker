# create the network (only need to do this once per docker)
docker network create -d bridge demo-net

# Remove any existing containers
docker rm -f httpA httpB httpC httpD httpE httpF httpG httpH httpI httpJ

# Remove associated volumes (if any)
docker volume rm httpA-mysql-data httpA-files httpA-index httpA-eiou httpA-backups
docker volume rm httpB-mysql-data httpB-files httpB-index httpB-eiou httpB-backups
docker volume rm httpC-mysql-data httpC-files httpC-index httpC-eiou httpC-backups

# Build base image
docker build -f eiou.dockerfile -t eiou/eiou .

# Start containers
