# create the network (only need to do this once per docker)
docker network create -d bridge demo-net

# Remove any existing containers
docker rm -f httpA httpB httpC httpD httpE httpF httpG httpH httpI httpJ

# Remove associated volumes (if any)
docker volume rm httpA-mysql-data httpA-config httpA-backups httpA-letsencrypt
docker volume rm httpB-mysql-data httpB-config httpB-backups httpB-letsencrypt
docker volume rm httpC-mysql-data httpC-config httpC-backups httpC-letsencrypt
docker volume rm httpD-mysql-data httpD-config httpD-backups httpD-letsencrypt
docker volume rm httpE-mysql-data httpE-config httpE-backups httpE-letsencrypt

docker volume rm httpJ-mysql-data httpJ-config httpJ-backups httpJ-letsencrypt

# Build base image
docker build -f eiou.dockerfile -t eiou/eiou .

# Start containers
docker run -d --restart unless-stopped  --network=demo-net --name httpA -v "httpA-mysql-data:/var/lib/mysql" -v "httpA-config:/etc/eiou/config" -v "httpA-backups:/var/lib/eiou/backups" -v "httpA-letsencrypt:/etc/letsencrypt" -e QUICKSTART=httpA eiou/eiou
