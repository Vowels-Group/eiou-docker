# Create the network (only need to do this once per docker installation)
docker network create -d bridge demo-net

# Remove any existing containers
powershell -Command "if (docker ps -aq) { docker rm -f $(docker ps -aq) } else { echo 'No containers to remove' }"

# Build base image
docker build -f eiou.dockerfile -t eiou .

# Start containers
docker run -d --network=demo-net --name A0 -e QUICKSTART=A0 eiou/eiou
docker run -d --network=demo-net --name A1 -e QUICKSTART=A1 eiou/eiou
docker run -d --network=demo-net --name A11 -e QUICKSTART=A11 eiou/eiou
docker run -d --network=demo-net --name A12 -e QUICKSTART=A12 eiou/eiou
docker run -d --network=demo-net --name A2 -e QUICKSTART=A2 eiou/eiou
docker run -d --network=demo-net --name A21 -e QUICKSTART=A21 eiou/eiou
docker run -d --network=demo-net --name A22 -e QUICKSTART=A22 eiou/eiou
docker run -d --network=demo-net --name A3 -e QUICKSTART=A3 eiou/eiou
docker run -d --network=demo-net --name A31 -e QUICKSTART=A31 eiou/eiou
docker run -d --network=demo-net --name A32 -e QUICKSTART=A32 eiou/eiou
docker run -d --network=demo-net --name A4 -e QUICKSTART=A4 eiou/eiou
docker run -d --network=demo-net --name A41 -e QUICKSTART=A41 eiou/eiou
docker run -d --network=demo-net --name A42 -e QUICKSTART=A42 eiou/eiou
sleep 5
