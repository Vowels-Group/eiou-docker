# create the network (only need to do this once per docker installation)
docker network create -d bridge demo-net

# Remove any existing containers
docker rm -f Alice Bob Carol Daniel

# Remove associated volumes (if any)
docker volume rm Alice-mysql-data Alice-config Alice-backups Alice-letsencrypt
docker volume rm Bob-mysql-data Bob-config Bob-backups Bob-letsencrypt
docker volume rm Carol-mysql-data Carol-config Carol-backups Carol-letsencrypt
docker volume rm Daniel-mysql-data Daniel-config Daniel-backups Daniel-letsencrypt

# Build base image
docker build -f eiou.dockerfile -t eiou/eiou .

# Start containers
docker run -d --restart unless-stopped  --network=demo-net --name Alice -v "Alice-mysql-data:/var/lib/mysql" -v "Alice-config:/etc/eiou/config" -v "Alice-backups:/var/lib/eiou/backups" -v "Alice-letsencrypt:/etc/letsencrypt" -e QUICKSTART=Alice eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name Bob -v "Bob-mysql-data:/var/lib/mysql" -v "Bob-config:/etc/eiou/config" -v "Bob-backups:/var/lib/eiou/backups" -v "Bob-letsencrypt:/etc/letsencrypt" -e QUICKSTART=Bob eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name Carol -v "Carol-mysql-data:/var/lib/mysql" -v "Carol-config:/etc/eiou/config" -v "Carol-backups:/var/lib/eiou/backups" -v "Carol-letsencrypt:/etc/letsencrypt" -e QUICKSTART=Carol eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name Daniel -v "Daniel-mysql-data:/var/lib/mysql" -v "Daniel-config:/etc/eiou/config" -v "Daniel-backups:/var/lib/eiou/backups" -v "Daniel-letsencrypt:/etc/letsencrypt" -e QUICKSTART=Daniel eiou/eiou
sleep 160 # Safety wait for all Tor connections to be properly ready

# add contacts
docker exec Alice eiou contact add http://Bob Bob --fee 0.1 --credit 1000 --currency USD
docker exec Bob eiou contact add http://Alice Alice --fee 0.1 --credit 1000 --currency USD
docker exec Bob eiou contact add http://Carol Carol --fee 0.1 --credit 1000 --currency USD
docker exec Carol eiou contact add http://Bob Bob --fee 0.1 --credit 1000 --currency USD
docker exec Carol eiou contact add http://Daniel Daniel --fee 0.1 --credit 1000 --currency USD
docker exec Daniel eiou contact add http://Carol Carol --fee 0.1 --credit 1000 --currency USD

# direct transaction
docker exec Alice eiou send http://Bob 100 USD
docker exec Alice eiou send Bob 40 USD

# p2p transaction
docker exec Alice eiou send http://Carol 60 USD
