# create the network (only need to do this once per docker installation)
docker network create -d bridge demo-net

# Remove any existing containers
docker rm -f Alice Bob Carol Daniel

# Remove associated volumes (if any)
docker volume rm Alice-mysql-data Alice-files Alice-index Alice-eiou Alice-backups
docker volume rm Bob-mysql-data Bob-files Bob-index Bob-eiou Bob-backups
docker volume rm Carol-mysql-data Carol-files Carol-index Carol-eiou Carol-backups
docker volume rm Daniel-mysql-data Daniel-files Daniel-index Daniel-eiou Daniel-backups

# Build base image
docker build -f eiou.dockerfile -t eiou/eiou .

# Start containers
docker run -d  --network=demo-net --name Alice -v "Alice-mysql-data:/var/lib/mysql" -v "Alice-files:/etc/eiou/" -v "Alice-backups:/var/lib/eiou/backups" -e QUICKSTART=Alice eiou/eiou
docker run -d  --network=demo-net --name Bob -v "Bob-mysql-data:/var/lib/mysql" -v "Bob-files:/etc/eiou/" -v "Bob-backups:/var/lib/eiou/backups" -e QUICKSTART=Bob eiou/eiou
docker run -d  --network=demo-net --name Carol -v "Carol-mysql-data:/var/lib/mysql" -v "Carol-files:/etc/eiou/" -v "Carol-backups:/var/lib/eiou/backups" -e QUICKSTART=Carol eiou/eiou
docker run -d  --network=demo-net --name Daniel -v "Daniel-mysql-data:/var/lib/mysql" -v "Daniel-files:/etc/eiou/" -v "Daniel-backups:/var/lib/eiou/backups" -e QUICKSTART=Daniel eiou/eiou
sleep 5

# add contacts
docker exec Alice eiou add http://Bob Bob 0.1 1000 USD
docker exec Bob eiou add http://Alice Alice 0.1 1000 USD
docker exec Bob eiou add http://Carol Carol 0.1 1000 USD
docker exec Carol eiou add http://Bob Bob 0.1 1000 USD
docker exec Carol eiou add http://Daniel Daniel 0.1 1000 USD
docker exec Daniel eiou add http://Carol Carol 0.1 1000 USD

# direct transaction
docker exec Alice eiou send http://Bob 100 USD
docker exec Alice eiou send Bob 40 USD

# p2p transaction
docker exec Alice eiou send http://Carol 60 USD
