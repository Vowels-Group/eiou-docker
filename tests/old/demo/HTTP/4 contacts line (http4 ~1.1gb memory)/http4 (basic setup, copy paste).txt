# create the network (only need to do this once per docker installation)
docker network create -d bridge demo-net

# Remove any existing containers
docker rm -f http0 http1 http2 http3

# Remove associated volumes (if any)
docker volume rm http0-mysql-data http0-config http0-backups http0-letsencrypt
docker volume rm http1-mysql-data http1-config http1-backups http1-letsencrypt
docker volume rm http2-mysql-data http2-config http2-backups http2-letsencrypt
docker volume rm http3-mysql-data http3-config http3-backups http3-letsencrypt

# Build base image
docker build -f eiou.dockerfile -t eiou/eiou .

# Start containers
docker run -d --restart unless-stopped  --network=demo-net --name http0 -v "http0-mysql-data:/var/lib/mysql" -v "http0-config:/etc/eiou/config" -v "http0-backups:/var/lib/eiou/backups" -v "http0-letsencrypt:/etc/letsencrypt" -e QUICKSTART=http0 eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name http1 -v "http1-mysql-data:/var/lib/mysql" -v "http1-config:/etc/eiou/config" -v "http1-backups:/var/lib/eiou/backups" -v "http1-letsencrypt:/etc/letsencrypt" -e QUICKSTART=http1 eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name http2 -v "http2-mysql-data:/var/lib/mysql" -v "http2-config:/etc/eiou/config" -v "http2-backups:/var/lib/eiou/backups" -v "http2-letsencrypt:/etc/letsencrypt" -e QUICKSTART=http2 eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name http3 -v "http3-mysql-data:/var/lib/mysql" -v "http3-config:/etc/eiou/config" -v "http3-backups:/var/lib/eiou/backups" -v "http3-letsencrypt:/etc/letsencrypt" -e QUICKSTART=http3 eiou/eiou
sleep 160 # Safety wait for all Tor connections to be properly ready

# add contacts
docker exec http0 eiou add http://http1 http1 0.1 1000 USD
docker exec http1 eiou add http://http0 http0 0.1 1000 USD
docker exec http1 eiou add http://http2 http2 0.1 1000 USD
docker exec http2 eiou add http://http1 http1 0.1 1000 USD
docker exec http2 eiou add http://http3 http3 0.1 1000 USD
docker exec http3 eiou add http://http2 http2 0.1 1000 USD
