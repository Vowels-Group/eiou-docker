
# create the network (only need to do this once per docker)
docker network create -d bridge demo-net

# Remove any existing containers
powershell -Command "if (docker ps -aq) { docker rm -f $(docker ps -aq) } else { echo 'No containers to remove' }"

# Build base image
docker build -f eioud.dockerfile -t eioud .

# Start containers
docker run -d  --network=demo-net --name httpsA -e QUICKSTART=httpsA eioud
docker run -d  --network=demo-net --name httpsB -e QUICKSTART=httpsB eioud
docker run -d  --network=demo-net --name httpsC -e QUICKSTART=httpsC eioud
docker run -d  --network=demo-net --name httpsD -e QUICKSTART=httpsD eioud
docker run -d  --network=demo-net --name httpsE -e QUICKSTART=httpsE eioud
docker run -d  --network=demo-net --name httpsF -e QUICKSTART=httpsF eioud
docker run -d  --network=demo-net --name httpsG -e QUICKSTART=httpsG eioud
docker run -d  --network=demo-net --name httpsH -e QUICKSTART=httpsH eioud
docker run -d  --network=demo-net --name httpsI -e QUICKSTART=httpsI eioud
docker run -d  --network=demo-net --name httpsJ -e QUICKSTART=httpsJ eioud
sleep 5
