
# create the network (only need to do this once per docker)
docker network create -d bridge demo-net

# Remove any existing containers
docker rm -f httpsA httpsB httpsC httpsD httpsE httpsF httpsG httpsH httpsI httpsJ

# Remove associated volumes (if any)
docker volume rm httpsA-mysql-data httpsA-config httpsA-backups httpsA-letsencrypt
