
# create the network (only need to do this once per docker)
docker network create -d bridge demo-net

# Remove any existing containers
powershell -Command "if (docker ps -aq) { docker rm -f $(docker ps -aq) } else { echo 'No containers to remove' }"

# Build base image
docker build -f eioud.dockerfile -t eioud .

# Start containers
docker run -d  --network=demo-net --name httpsA -e QUICKSTART=httpsA eiou/eiou
docker run -d  --network=demo-net --name httpsB -e QUICKSTART=httpsB eiou/eiou
docker run -d  --network=demo-net --name httpsC -e QUICKSTART=httpsC eiou/eiou
docker run -d  --network=demo-net --name httpsD -e QUICKSTART=httpsD eiou/eiou
docker run -d  --network=demo-net --name httpsE -e QUICKSTART=httpsE eiou/eiou
docker run -d  --network=demo-net --name httpsF -e QUICKSTART=httpsF eiou/eiou
docker run -d  --network=demo-net --name httpsG -e QUICKSTART=httpsG eiou/eiou
docker run -d  --network=demo-net --name httpsH -e QUICKSTART=httpsH eiou/eiou
docker run -d  --network=demo-net --name httpsI -e QUICKSTART=httpsI eiou/eiou
docker run -d  --network=demo-net --name httpsJ -e QUICKSTART=httpsJ eiou/eiou
sleep 5
