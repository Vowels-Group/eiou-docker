# create the network (only need to do this once per docker installation)
docker network create -d bridge demo-net

# Remove any existing containers
powershell -Command "if (docker ps -aq) { docker rm -f $(docker ps -aq) } else { echo 'No containers to remove' }"

# Build base image
docker build -f eioud.dockerfile -t eioud .

# Start containers
docker run -d  --network=demo-net --name https0 -e QUICKSTART=https0 eiou/eiou
docker run -d  --network=demo-net --name https1 -e QUICKSTART=https1 eiou/eiou
docker run -d  --network=demo-net --name https2 -e QUICKSTART=https2 eiou/eiou
docker run -d  --network=demo-net --name https3 -e QUICKSTART=https3 eiou/eiou
sleep 5
