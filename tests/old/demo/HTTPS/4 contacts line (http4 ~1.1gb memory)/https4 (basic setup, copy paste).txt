# create the network (only need to do this once per docker installation)
docker network create -d bridge demo-net

# Remove any existing containers
docker rm -f https0 https1 https2 https3

# Remove associated volumes (if any)
docker volume rm https0-mysql-data https0-files https0-backups https0-letsencrypt
docker volume rm https1-mysql-data https1-files https1-backups https1-letsencrypt
docker volume rm https2-mysql-data https2-files https2-backups https2-letsencrypt
docker volume rm https3-mysql-data https3-files https3-backups https3-letsencrypt

# Build base image
docker build -f eiou.dockerfile -t eiou/eiou .

# Start containers
docker run -d --restart unless-stopped  --network=demo-net --name https0 -v "https0-mysql-data:/var/lib/mysql" -v "https0-files:/etc/eiou/" -v "https0-backups:/var/lib/eiou/backups" -v "https0-letsencrypt:/etc/letsencrypt" -e QUICKSTART=https0 eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name https1 -v "https1-mysql-data:/var/lib/mysql" -v "https1-files:/etc/eiou/" -v "https1-backups:/var/lib/eiou/backups" -v "https1-letsencrypt:/etc/letsencrypt" -e QUICKSTART=https1 eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name https2 -v "https2-mysql-data:/var/lib/mysql" -v "https2-files:/etc/eiou/" -v "https2-backups:/var/lib/eiou/backups" -v "https2-letsencrypt:/etc/letsencrypt" -e QUICKSTART=https2 eiou/eiou
docker run -d --restart unless-stopped  --network=demo-net --name https3 -v "https3-mysql-data:/var/lib/mysql" -v "https3-files:/etc/eiou/" -v "https3-backups:/var/lib/eiou/backups" -v "https3-letsencrypt:/etc/letsencrypt" -e QUICKSTART=https3 eiou/eiou
sleep 15

# add contacts
docker exec https0 eiou add https://https1 https1 0.1 1000 USD
docker exec https1 eiou add https://https0 https0 0.1 1000 USD
docker exec https1 eiou add https://https2 https2 0.1 1000 USD
docker exec https2 eiou add https://https1 https1 0.1 1000 USD
docker exec https2 eiou add https://https3 https3 0.1 1000 USD
docker exec https3 eiou add https://https2 https2 0.1 1000 USD
