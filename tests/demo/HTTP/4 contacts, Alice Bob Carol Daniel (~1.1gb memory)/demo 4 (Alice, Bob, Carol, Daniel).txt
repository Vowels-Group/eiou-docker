# create the network (only need to do this once per docker installation)
docker network create -d bridge demo-net

# Remove any existing containers
powershell -Command "if (docker ps -aq) { docker rm -f $(docker ps -aq) } else { echo 'No containers to remove' }"

# build and start containers
docker build -f eioud.dockerfile -t eioud .
docker run -d  --network=demo-net --name Alice eioud
docker run -d  --network=demo-net --name Bob eioud
docker run -d  --network=demo-net --name Carol eioud
docker run -d  --network=demo-net --name Daniel eioud
sleep 5

# generate pubkeys and set hostname
docker exec Alice eiou generate http://Alice
docker exec Bob eiou generate http://Bob
docker exec Carol eiou generate http://Carol
docker exec Daniel eiou generate http://Daniel

# add contacts
docker exec Alice eiou add http://Bob Bob 0.1 1000 USD
docker exec Bob eiou add http://Alice Alice 0.1 1000 USD
docker exec Bob eiou add http://Carol Carol 0.1 1000 USD
docker exec Carol eiou add http://Bob Bob 0.1 1000 USD
docker exec Carol eiou add http://Daniel Daniel 0.1 1000 USD
docker exec Daniel eiou add http://Carol Carol 0.1 1000 USD

# direct transaction
docker exec Alice eiou send http://Bob 100 USD
