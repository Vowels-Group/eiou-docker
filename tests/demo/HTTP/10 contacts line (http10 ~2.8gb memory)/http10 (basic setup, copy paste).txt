# create the network (only need to do this once per docker)
docker network create -d bridge demo-net

# Remove any existing containers
powershell -Command "if (docker ps -aq) { docker rm -f $(docker ps -aq) } else { echo 'No containers to remove' }"


# build and start containers
docker build -f .\eioud.dockerfile -t eioud .
docker run -d  --network=demo-net --name httpA eioud
docker run -d  --network=demo-net --name httpB eioud
docker run -d  --network=demo-net --name httpC eioud
docker run -d  --network=demo-net --name httpD eioud
docker run -d  --network=demo-net --name httpE eioud
docker run -d  --network=demo-net --name httpF eioud
docker run -d  --network=demo-net --name httpG eioud
docker run -d  --network=demo-net --name httpH eioud
docker run -d  --network=demo-net --name httpI eioud
docker run -d  --network=demo-net --name httpJ eioud
sleep 5

# generate pubkeys and set hostname
