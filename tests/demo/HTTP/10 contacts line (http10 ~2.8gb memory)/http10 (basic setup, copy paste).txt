# create the network (only need to do this once per docker)
docker network create -d bridge demo-net

# Remove any existing containers
powershell -Command "if (docker ps -aq) { docker rm -f $(docker ps -aq) } else { echo 'No containers to remove' }"


# build and start containers
docker build -f .\eioud.dockerfile -t eioud .
docker run -d  --network=demo-net --name http0 eioud
docker run -d  --network=demo-net --name http1 eioud
docker run -d  --network=demo-net --name http2 eioud
docker run -d  --network=demo-net --name http3 eioud
docker run -d  --network=demo-net --name http4 eioud
docker run -d  --network=demo-net --name http5 eioud
docker run -d  --network=demo-net --name http6 eioud
docker run -d  --network=demo-net --name http7 eioud
docker run -d  --network=demo-net --name http8 eioud
docker run -d  --network=demo-net --name http9 eioud
