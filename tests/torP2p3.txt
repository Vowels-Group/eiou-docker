# copy this into docker to build and launch 3 containers at once
docker build -f .\eioud.dockerfile -t eioud .
docker run -d -v eioud0tor:/var/www/html/ --name eioud0tor eioud
docker run -d -v eioud1tor:/var/www/html/ --name eioud1tor eioud
docker run -d -v eioud2tor:/var/www/html/ --name eioud2tor eioud

# copy this into docker to generate addresses
docker exec eioud0tor eiou generate torAddressOnly
docker exec eioud1tor eiou generate torAddressOnly
docker exec eioud2tor eiou generate torAddressOnly
