mysql -u root

use eiou

# Debug - basic message log
SELECT message FROM debug;

# Contacts - relationship info (excludes pubkey, pubkey_hash)
SELECT id, name, contact_id, status, fee_percent, credit_limit, currency, online_status, valid_chain, last_ping_at FROM contacts;

# Addresses - network addresses
SELECT id, http, tor FROM addresses;

# Balances - account balances
SELECT id, received, sent, currency FROM balances;

# P2P Requests - peer-to-peer routing (excludes keys, signatures, salt)
SELECT id, hash, time, amount, currency, my_fee_amount, request_level, max_request_level, sender_address, destination_address, description, incoming_txid, outgoing_txid, status, created_at, completed_at FROM p2p;

# RP2P Responses - responses to p2p requests (excludes keys, signatures)
SELECT id, hash, time, amount, currency, sender_address FROM rp2p;

# Transactions - transaction records (excludes keys, signatures, nonce)
SELECT id, status, tx_type, type, sender_address, receiver_address, initial_sender_address, end_recipient_address, amount, currency, description, timestamp, txid, previous_txid, memo FROM transactions;

# Message Delivery - delivery tracking (excludes payload)
SELECT id, message_type, message_id, recipient_address, delivery_stage, retry_count, last_error, created_at, updated_at FROM message_delivery;

# Dead Letter Queue - failed messages for review (excludes payload)
SELECT id, message_type, message_id, recipient_address, failure_reason, status, retry_count, created_at, resolved_at FROM dead_letter_queue;

# Delivery Metrics - delivery statistics
SELECT id, message_type, period_start, period_end, total_sent, total_delivered, total_failed, avg_delivery_time_ms, avg_retry_count FROM delivery_metrics;

# API Keys - external API access (excludes encrypted_secret)
SELECT id, key_id, name, permissions, rate_limit_per_minute, enabled, created_at, last_used_at, expires_at FROM api_keys;

# API Request Log - API audit trail
SELECT id, key_id, endpoint, method, ip_address, request_timestamp, response_code, response_time_ms FROM api_request_log;

# Rate Limits - abuse prevention
SELECT id, identifier, action, attempts, first_attempt, last_attempt, blocked_until FROM rate_limits;

# Held Transactions - transactions pending resync
SELECT id, contact_pubkey_hash, txid, original_previous_txid, expected_previous_txid, transaction_type, hold_reason, sync_status, retry_count, held_at, resolved_at FROM held_transactions;
