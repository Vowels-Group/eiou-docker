mysql -u root

use eiou

# Debug - basic message log
SELECT message FROM debug;

# Contacts - relationship info
SELECT name, status, online_status, valid_chain, remote_version FROM contacts;

# Contact Currencies - per-contact currency settings
SELECT pubkey_hash, currency, fee_percent, credit_limit_whole, credit_limit_frac, status, direction FROM contact_currencies;

# Addresses - network addresses
SELECT http, https, tor FROM addresses;

# Balances - account balances
SELECT received_whole, received_frac, sent_whole, sent_frac, currency FROM balances;

# P2P Requests - peer-to-peer routing
SELECT hash, amount_whole, amount_frac, currency, my_fee_amount_whole, my_fee_amount_frac, rp2p_amount_whole, rp2p_amount_frac, request_level, max_request_level, sender_address, destination_address, fast, status, contacts_sent_count, contacts_responded_count, incoming_txid, outgoing_txid FROM p2p;

# RP2P Responses - responses to p2p requests
SELECT hash, amount_whole, amount_frac, currency, sender_address FROM rp2p;

# RP2P Candidates - best-fee route selection
SELECT hash, amount_whole, amount_frac, currency, sender_address, fee_amount_whole, fee_amount_frac FROM rp2p_candidates;

# P2P Senders - multi-path upstream senders per P2P hash
SELECT hash, sender_address FROM p2p_senders;

# P2P Relayed Contacts - contacts that returned already_relayed
SELECT hash, contact_address FROM p2p_relayed_contacts;

# Route Cancellations - cancel messages sent to losing P2P candidates
SELECT hash, contact_address, status FROM route_cancellations;

# Capacity Reservations - credit reserved during P2P routing
SELECT hash, contact_pubkey_hash, base_amount_whole, base_amount_frac, total_amount_whole, total_amount_frac, currency, status, release_reason FROM capacity_reservations;

# Chain Drop Proposals - mutual agreements to drop missing transactions
SELECT proposal_id, contact_pubkey_hash, missing_txid, broken_txid, direction, status FROM chain_drop_proposals;

# Transactions - transaction records
SELECT status, tx_type, type, sender_address, receiver_address, amount_whole, amount_frac, currency, description, txid, previous_txid, memo, needs_manual_review FROM transactions;

# Message Delivery - delivery tracking
SELECT message_type, message_id, recipient_address, delivery_stage, retry_count, last_error FROM message_delivery;

# Dead Letter Queue - failed messages for review
SELECT message_type, message_id, recipient_address, failure_reason, status, retry_count FROM dead_letter_queue;

# Delivery Metrics - delivery statistics
SELECT message_type, total_sent, total_delivered, total_failed, avg_delivery_time_ms, avg_retry_count FROM delivery_metrics;

# API Keys - external API access
SELECT key_id, name, permissions, rate_limit_per_minute, enabled FROM api_keys;

# API Nonces - replay protection for API requests
SELECT nonce, key_id FROM api_nonces;

# API Request Log - API audit trail
SELECT key_id, endpoint, method, ip_address, response_code, response_time_ms FROM api_request_log;

# Rate Limits - abuse prevention
SELECT identifier, action, attempts, blocked_until FROM rate_limits;

# Held Transactions - transactions pending resync
SELECT contact_pubkey_hash, txid, transaction_type, hold_reason, sync_status, retry_count FROM held_transactions;

# Contact Credit - available credit per contact
SELECT pubkey_hash, available_credit_whole, available_credit_frac, currency FROM contact_credit;
