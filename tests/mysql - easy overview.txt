mysql -u root

use eiou

# Debug - basic message log
SELECT message FROM debug;

# Contacts - relationship info (excludes pubkey, pubkey_hash)
SELECT id, name, contact_id, status, online_status, valid_chain, created_at, last_ping_at FROM contacts;

# Contact Currencies - per-contact currency settings (fee, credit limit, direction)
SELECT id, pubkey_hash, currency, fee_percent, credit_limit, status, direction, created_at, updated_at FROM contact_currencies;

# Addresses - network addresses (http, https, tor)
SELECT id, http, https, tor FROM addresses;

# Balances - account balances
SELECT id, received, sent, currency FROM balances;

# P2P Requests - peer-to-peer routing (excludes keys, signatures, salt)
SELECT id, hash, time, expiration, amount, currency, my_fee_amount, rp2p_amount, request_level, max_request_level, sender_address, destination_address, description, fast, contacts_sent_count, contacts_responded_count, hop_wait, contacts_relayed_count, contacts_relayed_responded_count, phase1_sent, status, sending_started_at, sending_worker_pid, incoming_txid, outgoing_txid, created_at, completed_at FROM p2p;

# RP2P Responses - responses to p2p requests (excludes keys, signatures)
SELECT id, hash, time, amount, currency, sender_address, created_at FROM rp2p;

# RP2P Candidates - best-fee route selection (excludes keys, signatures)
SELECT id, hash, time, amount, currency, sender_address, fee_amount, created_at FROM rp2p_candidates;

# P2P Senders - multi-path upstream senders per P2P hash (excludes pubkey)
SELECT id, hash, sender_address, created_at FROM p2p_senders;

# P2P Relayed Contacts - contacts that returned already_relayed (two-phase selection)
SELECT id, hash, contact_address, created_at FROM p2p_relayed_contacts;

# Route Cancellations - cancel messages sent to losing P2P candidates
SELECT id, hash, candidate_id, contact_address, status, created_at, acknowledged_at FROM route_cancellations;

# Capacity Reservations - credit reserved during P2P routing
SELECT id, hash, contact_pubkey_hash, base_amount, total_amount, currency, status, created_at, released_at, release_reason FROM capacity_reservations;

# Chain Drop Proposals - mutual agreements to drop missing transactions
SELECT id, proposal_id, contact_pubkey_hash, missing_txid, broken_txid, previous_txid_before_gap, direction, status, created_at, updated_at, expires_at, resolved_at FROM chain_drop_proposals;

# Transactions - transaction records (excludes keys, signatures, nonce, signed_message_content)
SELECT id, status, tx_type, type, sender_address, receiver_address, initial_sender_address, end_recipient_address, amount, currency, description, timestamp, txid, previous_txid, memo, sending_started_at, recovery_count, needs_manual_review, expires_at FROM transactions;

# Message Delivery - delivery tracking (excludes payload)
SELECT id, message_type, message_id, recipient_address, delivery_stage, retry_count, max_retries, next_retry_at, last_error, created_at, updated_at FROM message_delivery;

# Dead Letter Queue - failed messages for review (excludes payload)
SELECT id, message_type, message_id, recipient_address, failure_reason, status, retry_count, last_retry_at, created_at, resolved_at FROM dead_letter_queue;

# Delivery Metrics - delivery statistics
SELECT id, message_type, period_start, period_end, total_sent, total_delivered, total_failed, avg_delivery_time_ms, avg_retry_count, created_at FROM delivery_metrics;

# API Keys - external API access (excludes encrypted_secret)
SELECT id, key_id, name, permissions, rate_limit_per_minute, enabled, created_at, last_used_at, expires_at FROM api_keys;

# API Nonces - replay protection for API requests
SELECT nonce, key_id, created_at FROM api_nonces;

# API Request Log - API audit trail
SELECT id, key_id, endpoint, method, ip_address, request_timestamp, response_code, response_time_ms FROM api_request_log;

# Rate Limits - abuse prevention
SELECT id, identifier, action, attempts, first_attempt, last_attempt, blocked_until FROM rate_limits;

# Held Transactions - transactions pending resync
SELECT id, contact_pubkey_hash, txid, original_previous_txid, expected_previous_txid, transaction_type, hold_reason, sync_status, retry_count, max_retries, held_at, last_sync_attempt, next_retry_at, resolved_at FROM held_transactions;

# Contact Credit - available credit per contact (updated during ping)
SELECT id, pubkey_hash, available_credit, currency, updated_at FROM contact_credit;
