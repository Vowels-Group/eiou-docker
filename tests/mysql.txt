# mysql commands

# start mysql and enter db
mysql -u root
use eiou

# ============================================================
# DEBUG TABLE
# ============================================================
# Full debug log with context
SELECT id, timestamp, level, message, context, file, line, trace FROM debug;

# Quick message view
SELECT message FROM debug;

# ============================================================
# CONTACTS TABLE
# ============================================================
# Full contacts info (excludes pubkey, pubkey_hash)
# Note: fee_percent, credit_limit, and currency have moved to contact_currencies
SELECT id, contact_id, name, status, online_status, valid_chain, remote_version, created_at, last_ping_at FROM contacts;

# ============================================================
# CONTACT_CURRENCIES TABLE
# ============================================================
# Per-contact currency settings (fee, credit limit, direction)
# Replaced the old fee_percent/credit_limit/currency columns on contacts
SELECT id, pubkey_hash, currency, fee_percent, credit_limit_whole, credit_limit_frac, status, direction, created_at, updated_at FROM contact_currencies;

# ============================================================
# ADDRESSES TABLE
# ============================================================
# Network addresses (excludes pubkey_hash)
SELECT id, http, https, tor FROM addresses;

# ============================================================
# BALANCES TABLE
# ============================================================
# Account balances (excludes pubkey_hash)
SELECT id, received_whole, received_frac, sent_whole, sent_frac, currency FROM balances;

# ============================================================
# TRANSACTIONS TABLE
# ============================================================
# Full transactions (excludes pubkey, pubkey_hash, signatures, nonce, signed_message_content)
SELECT id, tx_type, type, status, sender_address, receiver_address, initial_sender_address, end_recipient_address, amount_whole, amount_frac, currency, timestamp, time, txid, previous_txid, memo, description, sending_started_at, recovery_count, needs_manual_review, expires_at FROM transactions;

# ============================================================
# P2P TABLE
# ============================================================
# Peer-to-peer requests (excludes pubkey, signatures, salt)
SELECT id, hash, time, expiration, currency, amount_whole, amount_frac, my_fee_amount_whole, my_fee_amount_frac, rp2p_amount_whole, rp2p_amount_frac, destination_address, request_level, max_request_level, sender_address, description, fast, contacts_sent_count, contacts_responded_count, hop_wait, contacts_relayed_count, contacts_relayed_responded_count, phase1_sent, status, sending_started_at, sending_worker_pid, created_at, incoming_txid, outgoing_txid, completed_at FROM p2p;

# ============================================================
# RP2P TABLE
# ============================================================
# Responses to p2p requests (excludes pubkey, signatures)
SELECT id, hash, time, amount_whole, amount_frac, currency, sender_address, created_at FROM rp2p;

# ============================================================
# RP2P_CANDIDATES TABLE
# ============================================================
# Best-fee route selection candidates (excludes pubkey, signatures)
SELECT id, hash, time, amount_whole, amount_frac, currency, sender_address, fee_amount_whole, fee_amount_frac, created_at FROM rp2p_candidates;

# ============================================================
# P2P_SENDERS TABLE
# ============================================================
# Multi-path upstream senders per P2P hash (excludes pubkey)
SELECT id, hash, sender_address, created_at FROM p2p_senders;

# ============================================================
# P2P_RELAYED_CONTACTS TABLE
# ============================================================
# Contacts that returned already_relayed during broadcast (two-phase selection)
SELECT id, hash, contact_address, created_at FROM p2p_relayed_contacts;

# ============================================================
# ROUTE_CANCELLATIONS TABLE
# ============================================================
# Cancel messages sent to losing P2P candidates after best-fee selection
SELECT id, hash, candidate_id, contact_address, status, created_at, acknowledged_at FROM route_cancellations;

# ============================================================
# CAPACITY_RESERVATIONS TABLE
# ============================================================
# Credit capacity reserved during P2P routing (released on cancel/expire/commit)
SELECT id, hash, contact_pubkey_hash, base_amount_whole, base_amount_frac, total_amount_whole, total_amount_frac, currency, status, created_at, released_at, release_reason FROM capacity_reservations;

# ============================================================
# CHAIN_DROP_PROPOSALS TABLE
# ============================================================
# Mutual agreements to drop missing transactions
SELECT id, proposal_id, contact_pubkey_hash, missing_txid, broken_txid, previous_txid_before_gap, direction, status, gap_context, created_at, updated_at, expires_at, resolved_at FROM chain_drop_proposals;

# ============================================================
# API_KEYS TABLE
# ============================================================
# API keys for external access (excludes encrypted_secret)
SELECT id, key_id, name, permissions, rate_limit_per_minute, enabled, created_at, last_used_at, expires_at FROM api_keys;

# ============================================================
# API_NONCES TABLE
# ============================================================
# Nonce replay protection for API requests (auto-cleaned after expiry)
SELECT nonce, key_id, created_at FROM api_nonces;

# ============================================================
# API_REQUEST_LOG TABLE
# ============================================================
# API request audit trail
SELECT id, key_id, endpoint, method, ip_address, request_timestamp, response_code, response_time_ms FROM api_request_log;

# ============================================================
# MESSAGE_DELIVERY TABLE
# ============================================================
# Message delivery tracking (excludes payload)
SELECT id, message_type, message_id, recipient_address, delivery_stage, retry_count, max_retries, next_retry_at, last_error, last_response, created_at, updated_at FROM message_delivery;

# ============================================================
# DEAD_LETTER_QUEUE TABLE
# ============================================================
# Failed messages for manual review (excludes payload)
SELECT id, message_type, message_id, recipient_address, retry_count, last_retry_at, failure_reason, status, created_at, resolved_at FROM dead_letter_queue;

# ============================================================
# DELIVERY_METRICS TABLE
# ============================================================
# Message delivery statistics
SELECT id, period_start, period_end, message_type, total_sent, total_delivered, total_failed, avg_delivery_time_ms, avg_retry_count, created_at FROM delivery_metrics;

# ============================================================
# RATE_LIMITS TABLE
# ============================================================
# Rate limiting for abuse prevention
SELECT id, identifier, action, attempts, first_attempt, last_attempt, blocked_until FROM rate_limits;

# ============================================================
# HELD_TRANSACTIONS TABLE
# ============================================================
# Transactions pending resync completion
SELECT id, contact_pubkey_hash, txid, original_previous_txid, expected_previous_txid, transaction_type, hold_reason, sync_status, retry_count, max_retries, held_at, last_sync_attempt, next_retry_at, resolved_at FROM held_transactions;

# ============================================================
# CONTACT_CREDIT TABLE
# ============================================================
# Available credit per contact (updated during ping)
SELECT id, pubkey_hash, available_credit_whole, available_credit_frac, currency, updated_at FROM contact_credit;
